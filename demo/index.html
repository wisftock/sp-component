Command Palette <sp-command-palette>

Spotlight-style command search triggered by keyboard shortcut.

Open (⌘K)
Show code
<sp-command-palette id="cmd"></sp-command-palette>

<script>
  const cmd = document.querySelector('#cmd');
  cmd.commands = [
    { id: 'new',      label: 'New file',     icon: '📄', shortcut: '⌘N' },
    { id: 'open',     label: 'Open file',    icon: '📂', shortcut: '⌘O' },
    { id: 'settings', label: 'Settings',     icon: '⚙️', shortcut: '⌘,' },
    { id: 'theme',    label: 'Toggle theme', icon: '🌙' },
  ];

  // Open on ⌘K / Ctrl+K
  window.addEventListener('keydown', (e) => {
    if ((e.metaKey || e.ctrlKey) && e.key === 'k') {
      e.preventDefault();
      cmd.open = true;
    }
  });
</script>
